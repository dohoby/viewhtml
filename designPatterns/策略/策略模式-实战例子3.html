    public String handle(int count) {
        //TODO 1分钟从队列里取60条数据
        StringBuilder resultMsg = new StringBuilder();

        for (int i = 0; i < count; i++) {
            try {
                TrademeWaitPostData trademeWaitPostData = getTrademeWaitPostData();
                if (trademeWaitPostData == null) {
                    log.info("队列没有数据");
                    resultMsg.append("队列没有数据");
                    break;
                }

                TrademeHandler trademeHandler = getTrademeHandler(trademeWaitPostData.getType());
                if (trademeHandler == null) {
                    log.error("没有此类型处理器");
                    resultMsg.append("没有此类型处理器");
                    continue;
                }

                if (trademeHandler instanceof TrademeDownListHandler) {

                                savePostDataOnFirst(trademeWaitPostData);//将失败的放回队列
                                resultMsg.append(handleMsg);
                            }
                        }
                    });
                } else {
                    boolean flag = excute(trademeHandler, trademeWaitPostData, resultMsg);
                    if (!flag) {
                        log.info("达到api调用限制");
                        break;
                    }
                }
            } catch (Exception e) {
                log.error("", e);
            }

        }

    public boolean excute(TrademeHandler trademeHandler, TrademeWaitPostData trademeWaitPostData, StringBuilder resultMsg) {

        String handleMsg = trademeHandler.handleResponse(response2);

        trademeHandler.handleTrademeCallbackData(trademeWaitPostData, response2, handleMsg);
        if (!StringUtils.isEmpty(handleMsg) && handleMsg.contains("Your API throttle limit has been exceeded")) {
            savePostDataOnFirst(trademeWaitPostData);//将失败的放回队列
            resultMsg.append(handleMsg);
            return false;
        }
        return true;
    }

    public String handleResponse(String response) {
        if (!StringUtils.isEmpty(response)) {
            if (response.startsWith("{")) {
                return "成功";
            } else {
                if (response.length() < 255) {
                    return response;
                } else {
                    return response.substring(0, 253);
                }
            }
        } else {
            return "trademe返回空";
        }
    }

    public String getUrl(String traceUrl, Map<String, String> paramMap) {
        traceUrl = traceUrl + "oauth_consumer_key={oauth_consumer_key}&oauth_token={oauth_token}&oauth_signature_method=HMAC-SHA1&oauth_timestamp={oauth_timestamp}&oauth_nonce={oauth_nonce}&oauth_signature={oauth_signature}&oauth_version=1.0";
        traceUrl = traceUrl.replace("{oauth_consumer_key}", paramMap.get("oauth_consumer_key"));
        traceUrl = traceUrl.replace("{oauth_token}", paramMap.get("oauth_token"));
        traceUrl = traceUrl.replace("{oauth_nonce}", paramMap.get("oauth_nonce"));
        traceUrl = traceUrl.replace("{oauth_signature}", paramMap.get("oauth_signature"));
        traceUrl = traceUrl.replace("{oauth_timestamp}", String.valueOf(paramMap.get("oauth_timestamp")));
        return traceUrl;
    }

}
