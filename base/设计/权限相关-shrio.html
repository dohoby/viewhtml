    private String extractUsername(String code) {

        try {
            OAuthClient oAuthClient = new OAuthClient(new URLConnectionClient());

            OAuthClientRequest accessTokenRequest = OAuthClientRequest.tokenLocation(oauth2Config.getAccessTokenUrl())
                    .setGrantType(GrantType.AUTHORIZATION_CODE).setClientId(oauth2Config.getClientId())
                    .setClientSecret(oauth2Config.getClientSecret()).setCode(code).setRedirectURI(oauth2Config.getFrontPageUrl())
                    .buildQueryMessage();
            String tmp = oauth2Config.getClientId() + ":" + oauth2Config.getClientSecret();
            accessTokenRequest.addHeader("Authorization", "Basic " + Base64.encodeToString(tmp.getBytes()));
            OAuthAccessTokenResponse oAuthResponse = oAuthClient.accessToken(accessTokenRequest, OAuth.HttpMethod.POST);

            String accessToken = oAuthResponse.getAccessToken();
            logger.info("OAuth2Realm accessToken:" + accessToken);
            Long expiresIn = oAuthResponse.getExpiresIn();
            logger.info("OAuth2Realm expiresIn:" + expiresIn);

            OAuthClientRequest userInfoRequest = new OAuthBearerClientRequest(oauth2Config.getUserInfoUrl())
                    .setAccessToken(accessToken).buildQueryMessage();

            OAuthResourceResponse resourceResponse = oAuthClient.resource(userInfoRequest, OAuth.HttpMethod.GET,
                    OAuthResourceResponse.class);
            String userInfo = resourceResponse.getBody();
            logger.info("OAuth2Realm userInfo:" + userInfo);
            JsonUtil.parseToMap(userInfo);
            JSONObject obj = new JSONObject(userInfo);
            String username = obj.getJSONObject("attributes").getString("Email");
            logger.info("OAuth2Realm obj:" + obj + ",username:" + username);
            return username;
        } catch (Exception e) {
            logger.error("Class: "+this.getClass().getName()+" method: "+
                    Thread.currentThread().getStackTrace()[1].getMethodName() +"Exception:{}", e);
            throw new OAuth2AuthenticationException(e);
        }
    }

OAuth2FormFilter extends FormAuthenticationFilter

package com.basic.config.oauth2;

import org.apache.shiro.subject.Subject;
import org.apache.shiro.web.filter.authc.FormAuthenticationFilter;
import org.apache.shiro.web.util.WebUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.http.HttpStatus;
import org.springframework.util.StringUtils;

import javax.annotation.Resource;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

public class OAuth2FormFilter extends FormAuthenticationFilter {
    private Logger logger = LoggerFactory.getLogger(OAuth2FormFilter.class);

    public static final String HEADER_LOCATION = "Location";
    public static final String HEADER_X_LOCATION = "X-Location";
    @Resource
    Oauth2Config oauth2Config;

    // oauth2 authc code参数名
    private String authcCodeParam = "code";

    public void setAuthcCodeParam(String authcCodeParam) {
        this.authcCodeParam = authcCodeParam;
    }

    @Override
    protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
//        logger.info("-----------------OAuth2FormFilter isAccessAllowed-----------------");
        return false;
    }

    @Override
    protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
        HttpServletRequest httpServletRequest = (HttpServletRequest) request;
        HttpServletResponse httpServletResponse = (HttpServletResponse) response;
        StringBuffer requestURL = httpServletRequest.getRequestURL();
//        logger.info("OAuth2FormFilter requestURL:" + requestURL);
        String host = requestURL.substring(0, requestURL.indexOf("/", 7));

        String requestUri = httpServletRequest.getRequestURI();
//        logger.info("OAuth2FormFilter requestUri:" + requestUri);

        String referer = httpServletRequest.getHeader("referer");
//        logger.info("OAuth2FormFilter referer:" + referer);
        // 原来只有上面一行，我加了下面的
        String requestedWith = httpServletRequest.getHeader("X-Requested-With");
//        logger.info("OAuth2FormFilter requestedWith:" + requestedWith);

        String method = httpServletRequest.getMethod();
        if ("OPTIONS".equalsIgnoreCase(method)) {
            return true;
        }

        Subject subject = getSubject(request, response);
        if (!subject.isAuthenticated()) {
//            logger.info("OAuth2FormFilter isAuthenticated:false");
            if (StringUtils.isEmpty(request.getParameter(authcCodeParam))) {
                // 如果用户没有身份验证，且没有auth code，则重定向到服务端授权
                if (!StringUtils.isEmpty(requestedWith) && requestedWith.equalsIgnoreCase("XMLHttpRequest")) {// 如果是ajax返回指定数据
//                    logger.info("------------------OAuth2FormFilter ajax请求--------------------");
                    httpServletResponse.setStatus(HttpStatus.FOUND.value());
                    if (StringUtils.isEmpty(referer)) {
                        httpServletResponse.addHeader(HEADER_X_LOCATION, getOaLoginUrl(host));
                    } else {
                        httpServletResponse.addHeader(HEADER_X_LOCATION, getOaLoginUrl(host) + "?pc_url=" + referer);
                    }
                    // saveRequest(request);不能保存请求，否则在OAuth2AuthenticationFilter会跳转到保存的该请求中，只能通过上面的pc_url来跳转
                } else {
//                    logger.info("------------------OAuth2FormFilter 浏览器请求--------------------");
                    // saveRequestAndRedirectToLogin(request, response);//
                    // 这里调用父类AccessControlFilter的getLoginUrl()方法获取application.yaml里的登录url来跳转，注：在ShiroConfig配置的
                    saveRequest(request);
                    redirectToLogin(request, response, host);
                }

                return false;
            }
        } else {
//            logger.info("OAuth2FormFilter isAuthenticated:true");
        }

        return true;
    }

    private String getOaLoginUrl(String host) {
        String oaLoginUrl = oauth2Config.getLoginUrl();
        oaLoginUrl = oaLoginUrl.replace("{{host}}", host);
//        logger.info("oaLoginUrl：" + oaLoginUrl);
        return oaLoginUrl;
    }

    protected void redirectToLogin(ServletRequest request, ServletResponse response, String host) throws IOException {
        String loginUrl = getLoginUrl();
//        logger.info("yaml配置文件LoginUrl：" + loginUrl);
        loginUrl = loginUrl.replace("{{host}}", host);
//        logger.info("yaml配置文件LoginUrl替换后：" + loginUrl);
        WebUtils.issueRedirect(request, response, loginUrl);
    }
}

package com.basic.config.oauth2;

import org.apache.shiro.authc.AuthenticationException;
import org.apache.shiro.authc.AuthenticationToken;
import org.apache.shiro.subject.Subject;

import java.util.Map;

public class OAuth2AuthenticationFilter extends AuthenticatingFilter {
    private Logger logger = LoggerFactory.getLogger(OAuth2AuthenticationFilter.class);
    @Resource
    Oauth2Config oauth2Config;

    // oauth2 authc code参数名
    private String authcCodeParam = "code";

    public void setAuthcCodeParam(String authcCodeParam) {
        this.authcCodeParam = authcCodeParam;
    }

    @Override
    protected AuthenticationToken createToken(ServletRequest request, ServletResponse response) throws Exception {
        HttpServletRequest httpRequest = (HttpServletRequest) request;
        String code = httpRequest.getParameter(authcCodeParam);
        return new OAuth2Token(code);
    }

    @Override
    protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
//        logger.info("-----------------OAuth2AuthenticationFilter isAccessAllowed-----------------");
        return false;
    }

    @Override
    protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
        HttpServletRequest httpServletRequest = (HttpServletRequest) request;
        HttpServletResponse httpServletResponse = (HttpServletResponse) response;
        String requestUri = httpServletRequest.getRequestURI();
//        logger.info("OAuth2AuthenticationFilter requestUri:" + requestUri);
        String referer = httpServletRequest.getHeader("referer");
        StringBuffer requestURL = httpServletRequest.getRequestURL();
        String host = requestURL.substring(0, requestURL.indexOf("/", 7));

        String method = httpServletRequest.getMethod();
        if ("OPTIONS".equalsIgnoreCase(method)) {
            return true;
        }

        Subject subject = getSubject(request, response);
        if (!subject.isAuthenticated()) {
//            logger.info("OAuth2AuthenticationFilter isAuthenticated:false");
//            logger.info("OAuth2AuthenticationFilter authcCodeParam：" + request.getParameter(authcCodeParam));
            Map<String, String[]> rmap = httpServletRequest.getParameterMap();
            for (String key : rmap.keySet()) {
//                logger.info("OAuth2AuthenticationFilter key：" + key + ",value:" + request.getParameter(key));
            }

            if (StringUtils.isEmpty(request.getParameter(authcCodeParam))) {
                // 如果用户没有身份验证，且没有auth code，则重定向到服务端授权
                // saveRequestAndRedirectToLogin(request, response);

                String requestedWith = httpServletRequest.getHeader("X-Requested-With");
//                logger.info("OAuth2AuthenticationFilter requestedWith:" + requestedWith);
                if (!StringUtils.isEmpty(requestedWith) && requestedWith.equalsIgnoreCase("XMLHttpRequest")) {// 如果是ajax返回指定数据
//                    logger.info("------------------ajax请求OAuth2AuthenticationFilter--------------------");
                    httpServletResponse.setStatus(HttpStatus.FOUND.value());
                    if (StringUtils.isEmpty(referer)) {
                        httpServletResponse.addHeader(OAuth2FormFilter.HEADER_X_LOCATION, getOaLoginUrl(host));
                    } else {
                        httpServletResponse.addHeader(OAuth2FormFilter.HEADER_X_LOCATION,
                                getOaLoginUrl(host) + "?pc_url=" + referer);
                    }
                } else {
//                    logger.info("------------------浏览器请求OAuth2AuthenticationFilter--------------------");
                    // saveRequestAndRedirectToLogin(request, response);//
                    // 这里调用父类AccessControlFilter的getLoginUrl()方法获取application.yaml里的登录url来跳转，注：在ShiroConfig配置的
                    saveRequest(request);
                    redirectToLogin(request, response, host);
                }

                return false;
            }
        } else {
//            logger.info("OAuth2AuthenticationFilter isAuthenticated:true");
            // saveRequestAndRedirectToLogin(request, response);
            // return false;
        }
        return executeLogin(request, response);
    }

    @Override
    protected boolean onLoginSuccess(AuthenticationToken token, Subject subject, ServletRequest request,
            ServletResponse response) throws Exception {
        HttpServletRequest httpServletRequest = (HttpServletRequest) request;
        String pcUrl = request.getParameter("pc_url");
        logger.info("OAuth2AuthenticationFilter onLoginSuccess pcUrl:" + pcUrl + ",uri:"
                + httpServletRequest.getRequestURI());
        if (!StringUtils.isEmpty(pcUrl)) {
            WebUtils.redirectToSavedRequest(request, response, pcUrl);
        } else {
            issueSuccessRedirect(request, response);
        }

        return false;
    }

    @Override
    protected boolean onLoginFailure(AuthenticationToken token, AuthenticationException ae, ServletRequest request,
            ServletResponse response) {
        // HttpServletResponse httpServletResponse = (HttpServletResponse) response;
        Subject subject = getSubject(request, response);
        if (subject.isAuthenticated() || subject.isRemembered()) {
            try {
                issueSuccessRedirect(request, response);
            } catch (Exception e) {
                logger.error("Class: "+this.getClass().getName()+" method: "+
                        Thread.currentThread().getStackTrace()[1].getMethodName() +"Exception:{}", e);
            }
        } else {
            try {
                WebUtils.issueRedirect(request, response, oauth2Config.getUnauthorizedUrl());
                // httpServletResponse.setStatus(HttpStatus.UNAUTHORIZED.value());
            } catch (Exception e) {
                logger.error("Class: "+this.getClass().getName()+" method: "+
                        Thread.currentThread().getStackTrace()[1].getMethodName() +"Exception:{}", e);
            }
        }
        return false;
    }

    private String getOaLoginUrl(String host) {
        String oaLoginUrl = oauth2Config.getLoginUrl();
        oaLoginUrl = oaLoginUrl.replace("{{host}}", host);
        logger.info("oaLoginUrl：" + oaLoginUrl);
        return oaLoginUrl;
    }

    protected void redirectToLogin(ServletRequest request, ServletResponse response, String host) throws IOException {
        String loginUrl = getLoginUrl();
        logger.info("yaml配置文件LoginUrl：" + loginUrl);
        loginUrl = loginUrl.replace("{{host}}", host);
        logger.info("yaml配置文件LoginUrl替换后：" + loginUrl);
        WebUtils.issueRedirect(request, response, loginUrl);
    }

     * Filter Chain定义说明 1、一个URL可以配置多个Filter，使用逗号分隔 2、当设置多个过滤器时，全部验证通过，才视为通过
     * 3、部分过滤器可指定参数，如perms，roles
     */
    @Bean
    public ShiroFilterFactoryBean shiroFilter(SecurityManager securityManager) {
        logger.info("ShiroConfiguration.shiroFilter()");
        ShiroFilterFactoryBean shiroFilterFactoryBean = new ShiroFilterFactoryBean();

        // 必须设置 SecurityManager
        shiroFilterFactoryBean.setSecurityManager(securityManager);
        shiroFilterFactoryBean.setLoginUrl(oauth2Config.getLoginUrl());

        // 登录成功后要跳转的链接
        shiroFilterFactoryBean.setSuccessUrl(oauth2Config.getFrontPageUrl());// 设置为前端首页
        // 未授权界面;
        shiroFilterFactoryBean.setUnauthorizedUrl("/unauthorizeError");

        shiroFilterFactoryBean.setFilterChainDefinitionMap(getDefaultFilterChainDefinitionMap());

        Map<String, Filter> filterMap = new LinkedHashMap<String, Filter>();
        filterMap.put("oauth2Authc", oAuth2AuthenticationFilter());
        // filterMap.put("authc", new
        // OAuth2FormFilter());//不能用下面的oAuth2FormFilter()注入bean，否则上面anon的配置无效;2、注意也不能直接new,OAuth2FormFilter有@Resource其他类
        filterMap.put(FILTER_NAME_AUTHC, oAuth2FormFilter());
        shiroFilterFactoryBean.setFilters(filterMap);

        return shiroFilterFactoryBean;
    }

    // <!-- 过滤链定义，从上向下顺序执行，一般将 /**放在最为下边 -->:这是一个坑呢，一不小心代码就不好使了;
    // <!-- authc:所有url都必须认证通过才可以访问; anon:所有url都都可以匿名访问-->
    private Map<String, String> getDefaultFilterChainDefinitionMap() {
        Map<String, String> filterChainDefinitionMap = new LinkedHashMap<String, String>();
        filterChainDefinitionMap.put("/public/**", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/logout", "logout");// 配置退出 过滤器,其中的具体的退出代码Shiro已经替我们实现了
        filterChainDefinitionMap.put("/login", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/logout", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/success", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/error", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/unauthorizeError", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/oauth2-login", "oauth2Authc");// 这样设置则只有访问oauth2-login才会调用这个过滤器
        filterChainDefinitionMap.put("/failureUrl", FILTER_NAME_ANON);
        filterChainDefinitionMap.put("/**", FILTER_NAME_AUTHC);
        return filterChainDefinitionMap;
    }

    /**
     * 这个是解决上面filterMap.put("authc", oAuth2FormFilter());自定义过滤器后，过滤失效的问题
     */
    @Bean
    public FilterRegistrationBean registration(OAuth2FormFilter filter) {
        FilterRegistrationBean registration = new FilterRegistrationBean(filter);
        registration.setEnabled(false);
        return registration;
    }

    @Bean(name = "oAuth2FormFilter")
    public OAuth2FormFilter oAuth2FormFilter() {
        OAuth2FormFilter OAuth2FormFilter = new OAuth2FormFilter();
        return OAuth2FormFilter;
    }

    @Bean(name = "oAuth2AuthenticationFilter")
    public OAuth2AuthenticationFilter oAuth2AuthenticationFilter() {
        OAuth2AuthenticationFilter oAuth2AuthenticationFilter = new OAuth2AuthenticationFilter();
        return oAuth2AuthenticationFilter;
    }

    @Bean(name = "configWebSessionManager")
    public DefaultWebSessionManager configWebSessionManager() {
        DefaultWebSessionManager manager = new DefaultWebSessionManager();
        // manager.setCacheManager(ehCacheManager());// 加入缓存管理器
        manager.setDeleteInvalidSessions(true);// 删除过期的session

        // 设置全局session超时时间，单位毫秒,必须要设置比redis的过期时间略小但不能太小，必然会导致内部登录引起闪屏
        manager.setGlobalSessionTimeout(1800000L);

        manager.setSessionValidationSchedulerEnabled(true);// 是否定时检查session

        return manager;
    }

    @Bean
    public SecurityManager securityManager() {
        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
        // 设置realm.
        securityManager.setRealm(getUserRealm());
        // 注入缓存管理器;
        // securityManager.setCacheManager(ehCacheManager());// 这个如果执行多次，也是同样的一个对象;
        // 注入记住我管理器;
        securityManager.setRememberMeManager(rememberMeManager());

        return securityManager;
    }

    /**
     * 身份认证realm; (这个需要自己写，账号密码校验；权限等)

    }

    }

    /**

package com.tobe.pmc.pc.config;

import java.io.IOException;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

@Component
public class CorsFilter implements Filter {
    private static final Logger logger = LoggerFactory.getLogger(CorsFilter.class);
    public static final String HEADERS = "Origin, No-Cache,Cache-Control, X-Requested-With, If-Modified-Since, Pragma, Last-Modified, Cache-Control, Expires, Content-Type, X-E4M-With,userId,token,Accept,Authorization,X-Location,Location";

    public void doFilter(ServletRequest req, ServletResponse res, FilterChain chain)
            throws IOException, ServletException {
        HttpServletResponse response = (HttpServletResponse) res;
        HttpServletRequest reqs = (HttpServletRequest) req;
        String origin = reqs.getHeader("Origin");
        response.setHeader("Access-Control-Allow-Origin", origin);
        response.setContentType("application/json;charset=UTF-8");
        response.setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS, DELETE");
        response.setHeader("Access-Control-Max-Age", "3600");
        response.setHeader("Access-Control-Allow-Headers", HEADERS);
        response.setHeader("Access-Control-Allow-Credentials", "true");
        response.setHeader("Access-Control-Expose-Headers", HEADERS);
        response.setHeader("XDomainRequestAllowed", "1");
//        logger.trace("*********************************过滤器被使用**************************");
        chain.doFilter(req, res);
    }

    public void init(FilterConfig filterConfig) {
    }

    public void destroy() {
    }

}
